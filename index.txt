// ─────────────────────────────────────────────────────────
// PROFILE CARD
// ─────────────────────────────────────────────────────────
function updateProfile(genName) {
    const p = genProfiles[genName];

    document.getElementById('profileName').textContent   = p.name;
    document.getElementById('profileSnappy').textContent = p.snappy;
    // Build descriptor chain: desc1 · desc2 · desc3 → desc4 · desc5
    const mid = Math.floor(p.descriptors.length / 2);
    const nowDescs  = p.descriptors.slice(0, mid + 1);
    const futDescs  = p.descriptors.slice(mid + 1);
    const sepHTML   = '<span class="desc-sep">&middot;</span>';
    const arrowHTML = '<span class="arrow" style="color:#0178f0;font-size:18px">&rarr;</span>';
    document.getElementById('journeyDescriptors').innerHTML =
        nowDescs.map(d => `<span class="descriptor">${d}</span>`).join(sepHTML)
        + arrowHTML
        + futDescs.map(d => `<span class="descriptor">${d}</span>`).join(sepHTML);
    document.getElementById('profileNow').textContent    = p.now;
    document.getElementById('profileFuture').textContent = p.future;

    // Stats
    document.getElementById('statsGrid').innerHTML =
        p.stats.map(s => `<div class="stat-pill">${s}</div>`).join('');
}

function showProfile(genName) {
    const card = document.getElementById('genProfile');
    card.classList.remove('active');
    setTimeout(() => {
        updateProfile(genName);
        card.classList.add('active');
    }, 270);
}

// GENERATION SLIDER
// ─────────────────────────────────────────────────────────
const genSlider = document.getElementById('genSlider');
const genLabelEls = document.querySelectorAll('.gen-label');

function setActiveGenLabel(index) {
    genLabelEls.forEach((el, i) => {
        el.classList.toggle('active', i === index);
    });
}

genSlider.addEventListener('input', function() {
    const newIndex = parseInt(this.value);
    if (newIndex === currentGenIndex) return;
    currentGenIndex = newIndex;
    setActiveGenLabel(newIndex);

    // Slide transition on chart
    const wrapper = document.getElementById('chartWrapper');
    wrapper.classList.add('transitioning');
    setTimeout(() => {
        updateChart(false);
        wrapper.classList.remove('transitioning');
    }, 160);

    showProfile(GENERATIONS[newIndex]);
});

// Clicking a label snaps slider to that generation
genLabelEls.forEach(el => {
    el.addEventListener('click', function() {
        const idx = parseInt(this.dataset.index);
        genSlider.value = idx;
        genSlider.dispatchEvent(new Event('input'));
    });
});
